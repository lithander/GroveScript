rotate 90+sin(time()*100)*150
set width, 3
set length, 1
seed Plant, Step(2,10)
repeat 9
	grow Plant, growth
	
grow Plant, resolveFoo
	
run Plant
	
#Step(a, b)
size width/(length^0.15)
visible 1/length^0.5
curve b, 5*a/length
move 1
raise length
out a

#resolveFoo
Foo(a,b) -> Step(a,b)

#growth
Step(a,b) : a > 4 -> Foo(a, b) Step(a, b+sin(time()*30)*60)
Step(a,b) -> Foo(a, b) Step(a+1, 60)
	out a, b
